package com.irurueta.numerical.interpolation;

/**
 * Computes linear interpolation.
 */
public class LinearInterpolator extends BaseInterpolator {

    /**
     * Length of x's and y's to take into account.
     */
    private static final int M = 2;

    public LinearInterpolator(final double[] x, final double[] y) {
        super(x, y, M);
    }

    /**
     * Actual interpolation method to be implemented by subclasses.
     *
     * @param j index where value x to be interpolated in located in the array of xx.
     * @param x value to obtain interpolation for.
     * @return interpolated value.
     */
    @Override
    public double rawinterp(int j, double x) {
        if (xx[j] == xx[j + 1]) {
            return yy[j];
        } else {
            return yy[j] + ((x - xx[j]) / (xx[j + 1] - xx[j])) * (yy[j + 1] - yy[j]);
        }
    }
}

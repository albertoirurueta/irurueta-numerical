16  package com.irurueta.numerical.integration;
17  
18  import com.irurueta.numerical.EvaluationException;
19  
20  /**
21   * Interface to define how single dimension functions can be evaluated in Double Exponential Rule
22   * Quadrature function integrators.
23   */
24  public interface DoubleExponentialSingleDimensionFunctionEvaluatorListener {
25  
26      /**
27       * Evaluates a single dimension function such as f(x) at provided point and returns the result.
28       *
29       * @param x     point where function will be evaluated.
30       * @param delta value to handle singularities. If the function has no singularities, or the
31       *              singularities are “mild” (e.g., no worse than logarithmic), this can be ignored.
32       * @return value returned by the function.
33       * @throws EvaluationException raised if something failed during the evaluation.
34       */
35      double evaluate(final double x, final double delta) throws EvaluationException;
36  }

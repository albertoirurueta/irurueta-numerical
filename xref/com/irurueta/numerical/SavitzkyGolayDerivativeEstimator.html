16  package com.irurueta.numerical;
17  
18  import com.irurueta.algebra.AlgebraException;
19  import com.irurueta.algebra.ArrayUtils;
20  import com.irurueta.algebra.Matrix;
21  import com.irurueta.algebra.SingularValueDecomposer;
22  import com.irurueta.algebra.Utils;
23  
24  /**
25   * Class to estimate the derivative of a single dimension function at a given
26   * point.
27   * The algorithm used in this implementation is valid for continuous functions
28   * only, otherwise inaccurate results might be obtained.
29   * This implementation is more robust against small discontinuities than
30   * SymmetricDerivativeEstimator, but it is also slower to compute.
31   * This method interpolates the sampled function values into a polynomial of
32   * 2nd degree (parabolic), whose derivative is known.
33   * Because a linear system of equations has to be solved to determine such
34   * polynomial, this method might be less accurate when large values are involved
35   * due to limited machine precision.
36   */
37  public class SavitzkyGolayDerivativeEstimator extends DerivativeEstimator {
38  
39      /**
40       * Number of required point to evaluate to compute derivative.
41       */
42      public static final int N_POINTS = 3;
43  
44      /**
45       * Constructor.
46       *
47       * @param listener listener to evaluate a single dimension function.
48       */
49      public SavitzkyGolayDerivativeEstimator(final SingleDimensionFunctionEvaluatorListener listener) {
50          super(listener);
51      }
52  
53      /**
54       * Computes the function derivative at provided point x.
55       *
56       * @param x Point where derivative is estimated.
57       * @return Derivative of function at provided point.
58       * @throws EvaluationException Raised if function cannot be properly
59       *                             evaluated.
60       */
61      @Override
62      @SuppressWarnings("Duplicates")
63      public double derivative(final double x) throws EvaluationException {
64          // fit a polynomial of degree 2 by evaluating function at x-h, x and x+h
65          var h = EPS * Math.abs(x);
66          if (h == 0.0) {
67              // Trick to reduce finite-precision error
68              h = EPS;
69          }
70  
71          final var xh1 = x + h;
72          final var xh2 = x - h;
73  
74          final var f = listener.evaluate(x);
75          final var fh1 = listener.evaluate(xh1);
76          final var fh2 = listener.evaluate(xh2);
77  
78          // express the problem as:
79          // a * x^2 + b * x + c = f(x)
80          // b * xh1^2 + b * xh1 + c = f(xh1)
81          // c * xh2^2 + b * xh2 + c = f(xh2)
82  
83          final double aParam;
84          final double bParam;
85          try {
86              final var a = new Matrix(N_POINTS, N_POINTS);
87  
88              a.setElementAt(0, 0, x * x);
89              a.setElementAt(1, 0, xh1 * xh1);
90              a.setElementAt(2, 0, xh2 * xh2);
91  
92              a.setElementAt(0, 1, x);
93              a.setElementAt(1, 1, xh1);
94              a.setElementAt(2, 1, xh2);
95  
96              a.setElementAt(0, 2, 1.0);
97              a.setElementAt(1, 2, 1.0);
98              a.setElementAt(2, 2, 1.0);
99  
100             final var b = new double[N_POINTS];
101 
102 
103             // normalize to increase accuracy
104             final var normA = Utils.normF(a);
105             a.multiplyByScalar(1.0 / normA);
106 
107             b[0] = f;
108             b[1] = fh1;
109             b[2] = fh2;
110 
111             // normalize to increase accuracy
112             ArrayUtils.multiplyByScalar(b, 1.0 / normA, b);
113 
114             final SingularValueDecomposer decomposer = new SingularValueDecomposer(a);
115 
116             decomposer.decompose();
117 
118             // now solve the system of equations in Least Mean Squared Error
119             // because SVD allows the system of equations to be solved using the
120             // pseudo-inverse
121             final double[] params = decomposer.solve(b);
122             aParam = params[0];
123             bParam = params[1];
124 
125         } catch (final AlgebraException e) {
126             return Double.NaN;
127         }
128 
129         // and c = params[2], but we don't need it
130 
131         // because we have fitted the function into a polynomial that has
132         // expression: a * x^2 + b * x + c, then its derivative is:
133         // 2.0 * a * x + b, therefore:
134         return 2.0 * aParam * x + bParam;
135     }
136 }

134     @Test
135     void integrate_whenGaussian_returnsExpectedResult() throws IntegrationException {
136         final var randomizer = new UniformRandomizer();
137         final var a = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
138         final var b = randomizer.nextDouble(a, MAX_VALUE);
139         final var mu = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
140         final var sigma = ABSOLUTE_ERROR_GAUSSIAN + Math.abs(randomizer.nextDouble(MIN_VALUE, MAX_VALUE));
141 
142         final var expected = NormalDist.cdf(b, mu, sigma) - NormalDist.cdf(a, mu, sigma);
143 
144         final var integrator = new RombergDoubleExponentialRuleQuadratureIntegrator(a, b,
145                 point -> NormalDist.p(point, mu, sigma));
146         final var result = integrator.integrate();
147 
148         assertEquals(expected, result, ABSOLUTE_ERROR_GAUSSIAN);
149     }

153         final var randomizer = new UniformRandomizer();
154         final var a = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
155         final var b = randomizer.nextDouble(a, MAX_VALUE);
156         final var lambda = randomizer.nextDouble(MIN_LAMBDA, MAX_LAMBDA);
157 
158         final var expected = 1.0 / lambda * (Math.exp(lambda * b) - Math.exp(lambda * a));

204     private static void assertPolynomialIntegration(final int degree, final double error) throws IntegrationException {
205         final var polynomial = buildPolynomial(degree);
206         final var integrationPolynomial = polynomial.integrationAndReturnNew();
207 
208         // set integration interval
209         final var randomizer = new UniformRandomizer();
210         final var a = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
211         final var b = randomizer.nextDouble(a, MAX_VALUE);
212 
213         final var expected = integrationPolynomial.evaluate(b) - integrationPolynomial.evaluate(a);
214 
215         final var integrator = new RombergDoubleExponentialRuleQuadratureIntegrator(a, b, polynomial::evaluate);
216         final var result = integrator.integrate();
217 
218         assertEquals(expected, result, error);
219     }
220 
221     private static Polynomial buildPolynomial(final int degree) {
222         final var randomizer = new UniformRandomizer();
223         final var result = new Polynomial(1.0);
224         for (var i = 0; i < degree; i++) {
225             final var root = randomizer.nextDouble(MIN_VALUE, MAX_VALUE);
226             final var poly = new Polynomial(-root, 1.0);
227             result.multiply(poly);
228         }
229 
230         return result;
231     }

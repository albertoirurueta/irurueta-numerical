268     @Test
269     void testEstimate() throws LockedException, NotReadyException, RobustEstimatorException {
270         for (var i = 0; i < TIMES; i++) {
271             final var randomizer = new UniformRandomizer();
272             final var numSamples = randomizer.nextInt(MIN_POINTS, MAX_POINTS);
273             final var listener = new TestPROMedSCRobustEstimatorListener(numSamples, PERCENTAGE_OUTLIER, THRESHOLD);
274             final var estimator = new PROMedSRobustEstimator<double[]>();
275 
276             // Force NotReadyException
277             assertThrows(NotReadyException.class, estimator::estimate);
278 
279             // set listener
280             estimator.setListener(listener);
281             listener.reset();
282             assertEquals(0, listener.getStartCounter());
283             assertEquals(0, listener.getEndCounter());
284             assertFalse(estimator.isLocked());
285 
286             // estimate
287             final var params = estimator.estimate();
288 
289             // check status after estimation
290             assertFalse(estimator.isLocked());
291             assertEquals(1, listener.getStartCounter());
292             assertEquals(1, listener.getEndCounter());
293 
294             // check correctness of estimation
295             assertEquals(params.length, listener.getParams().length);
296             assertEquals(NUM_PARAMS, params.length);
297 
298             assertArrayEquals(params, listener.getParams(), ABSOLUTE_ERROR);
299 
300             final var inliersData = estimator.getBestInliersData();
301             assertNotNull(inliersData);
302             assertSame(inliersData, estimator.getInliersData());
303             assertTrue(inliersData.getNumInliers() > 0);
304             assertNotNull(inliersData.getInliers());
305             assertNotNull(inliersData.getResiduals());
306         }
307     }

309     private static double[] computeParams() {
310         // we will estimate parameters a and b for equation y = a*x + b
311         final var randomizer = new UniformRandomizer();
312         final var params = new double[NUM_PARAMS];
313         // a parameter
314         params[0] = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
315         // b parameter
316         params[1] = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
317         return params;
318     }
319 
320     private static void computeSamplesAndQualityScores(
321             final double[] params, final int numSamples, final int percentageOutliers, final double[] ys,
322             final double[] xs, final double[] qualityScores) {
323 
324         final var randomizer = new UniformRandomizer();
325         for (var i = 0; i < numSamples; i++) {
326             // compute x values
327             xs[i] = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
328             // compute exact y values
329             ys[i] = params[0] * xs[i] + params[1];
330             final var scoreError = randomizer.nextDouble(MIN_SCORE_ERROR, MAX_SCORE_ERROR);
331             // inliers score can also have error
332             qualityScores[i] = 1.0 + scoreError;
333             if (randomizer.nextInt(0, 100) < percentageOutliers) {
334                 // is outlier, so we add a certain amount of error
335                 final var error = randomizer.nextDouble(MIN_ERROR, MAX_ERROR);
336                 // add sample error
337                 ys[i] += error;
338                 // quality score is (1 / (1 + error)) + scoreError
339                 qualityScores[i] = 1.0 / (1.0 + error) + scoreError;
340             }
341         }
342     }

344     private static class TestPROMedSCRobustEstimatorListener implements PROMedSRobustEstimatorListener<double[]> {
345 
346         private final double[] params;
347         private final double[] xs;
348         private final double[] ys;
349         private final double[] qualityScores;
350         private final int numSamples;
351         private final double threshold;
352 
353         private int startCounter;
354         private int endCounter;
355         private float previousProgress;
356 
357         TestPROMedSCRobustEstimatorListener(final int numSamples, final int percentageOutliers,
358                                             final double threshold) {
359             this.numSamples = numSamples;
360             params = computeParams();
361             xs = new double[numSamples];
362             ys = new double[numSamples];
363             qualityScores = new double[numSamples];
364             computeSamplesAndQualityScores(params, numSamples, percentageOutliers, ys, xs, qualityScores);
365             this.threshold = threshold;
366             reset();
367         }
368 
369         public double[] getParams() {
370             return params;
371         }
372 
373         @Override
374         public double[] getQualityScores() {
375             return qualityScores;
376         }
377 
378         @Override
379         public double getThreshold() {
380             return threshold;
381         }
382 
383         @Override
384         public int getTotalSamples() {
385             return numSamples;
386         }
387 
388         @Override
389         public int getSubsetSize() {
390             // only two matches x and y are required to determine a and b as
391             // follows:
392             // y1 = a* x1 + b --> b = y1 - a * x1
393             // y2 = a * x2 + b --> y2 = a * x2 + y1 - a * x1 -->
394             // y2 - y1 = (x2 - x1)*a --> a = (y2 - y1) / (x2 - x1)
395 
396             // Hence:
397             // a = (y2 - y1) / (x2 - x1)
398             // b = y1 - (y2 - y1) / (x2 - x1) * x1
399             return NUM_PARAMS;
400         }
401 
402         @Override
403         public void estimatePreliminarSolutions(final int[] samplesIndices, final List<double[]> solutions) {
404 
405             if (samplesIndices.length != NUM_PARAMS) {
406                 throw new IllegalArgumentException();
407             }
408             final var index1 = samplesIndices[0];
409             final var index2 = samplesIndices[1];
410 
411             final var y1 = ys[index1];
412             final var y2 = ys[index2];
413             final var x1 = xs[index1];
414             final var x2 = xs[index2];
415 
416             final var a = (y2 - y1) / (x2 - x1);
417             final var b = y1 - a * x1;
418 
419             final var solution = new double[NUM_PARAMS];
420             solution[0] = a;
421             solution[1] = b;
422 
423             solutions.add(solution);
424         }
425 
426         @Override
427         public double computeResidual(final double[] currentEstimation, final int i) {
428             final var a = currentEstimation[0];
429             final var b = currentEstimation[1];
430 
431             final var estimatedY = a * xs[i] + b;
432             final var y = ys[i];
433 
434             return Math.abs(estimatedY - y);
435         }
436 
437         @Override
438         public boolean isReady() {
439             return params != null && xs != null && ys != null;
440         }
441 
442         @Override
443         public void onEstimateStart(final RobustEstimator<double[]> estimator) {
444             testIsLocked((PROMedSRobustEstimator<double[]>) estimator);
445             startCounter++;
446         }
447 
448         @Override
449         public void onEstimateEnd(final RobustEstimator<double[]> estimator) {
450             testIsLocked((PROMedSRobustEstimator<double[]>) estimator);
451             endCounter++;
452         }
453 
454         @Override
455         public void onEstimateNextIteration(final RobustEstimator<double[]> estimator, final int iteration) {
456             final var ransacEstimator = (PROMedSRobustEstimator<double[]>) estimator;
457             testIsLocked(ransacEstimator);
458             assertTrue(iteration > 0);
459             assertTrue(ransacEstimator.getNIters() >= 0);
460             assertNotNull(ransacEstimator.getBestResult());
461         }
462 
463         @Override
464         public void onEstimateProgressChange(final RobustEstimator<double[]> estimator, final float progress) {
465             testIsLocked((PROMedSRobustEstimator<double[]>) estimator);
466             assertTrue(progress >= 0.0f);
467             assertTrue(progress <= 1.0f);
468             assertTrue(progress >= previousProgress);
469             previousProgress = progress;
470         }
471 
472         int getStartCounter() {
473             return startCounter;
474         }
475 
476         int getEndCounter() {
477             return endCounter;
478         }

480         private void testIsLocked(final PROMedSRobustEstimator<double[]> estimator) {
481             assertTrue(estimator.isLocked());
482             // test that estimator cannot be modified while locked
483             assertThrows(LockedException.class, () -> estimator.setConfidence(0.5));
484             assertThrows(LockedException.class, () -> estimator.setListener(this));
485             assertThrows(LockedException.class, () -> estimator.setMaxIterations(1));
486             assertThrows(LockedException.class, () -> estimator.setProgressDelta(0.5f));
487             assertThrows(LockedException.class, () -> estimator.setMaxOutliersProportion(0.5));
488             assertThrows(LockedException.class, () -> estimator.setEta0(0.5));
489             assertThrows(LockedException.class, () -> estimator.setBeta(0.5));
490             assertThrows(LockedException.class, () -> estimator.setStopThresholdEnabled(false));
491         }
492 
493         public final void reset() {
494             startCounter = endCounter = 0;
495             previousProgress = 0.0f;
496         }

16  package com.irurueta.numerical.robust;
17  
18  import com.irurueta.sorting.SortingException;
19  import com.irurueta.statistics.UniformRandomizer;
20  import org.junit.jupiter.api.Test;
21  
22  import static org.junit.jupiter.api.Assertions.*;
23  
24  class WeightSelectionTest {
25  
26      private static final int MIN_LENGTH = 6;
27      private static final int MAX_LENGTH = 50;
28  
29      private static final double MIN_RANDOM_VALUE = 0.0;
30      private static final double MAX_RANDOM_VALUE = 1.0;
31  
32      @Test
33      void testSelectWeights() throws SortingException {
34  
35          final var randomizer = new UniformRandomizer();
36          final var length = randomizer.nextInt(MIN_LENGTH, MAX_LENGTH);
37  
38          final var weights = new double[length];
39          randomizer.fill(weights, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
40  
41          // test with sorting disabled and num selected lower than length
42          var maxPoints = length - 1;
43          var selection = WeightSelection.selectWeights(weights, false, maxPoints);
44          assertEquals(selection.getNumSelected(), maxPoints);
45          // check first maxPoints values are true
46          for (var i = 0; i < maxPoints; i++) {
47              assertTrue(selection.getSelected()[i]);
48          }
49          for (var i = maxPoints; i < length; i++) {
50              assertFalse(selection.getSelected()[i]);
51          }
52  
53          // test with sorting disabled and num selected greater than length
54          maxPoints = length + 1;
55          selection = WeightSelection.selectWeights(weights, false, maxPoints);
56          assertEquals(selection.getNumSelected(), length);
57          // check all values are true
58          for (var i = 0; i < length; i++) {
59              assertTrue(selection.getSelected()[i]);
60          }
61  
62          // test with sorting enabled
63          maxPoints = length - 1;
64          selection = WeightSelection.selectWeights(weights, true, maxPoints);
65          assertEquals(selection.getNumSelected(), maxPoints);
66          // check all values are true
67          var counter = 0;
68          for (var i = 0; i < length; i++) {
69              if (selection.getSelected()[i]) {
70                  counter++;
71              }
72          }
73          assertEquals(counter, selection.getNumSelected());
74      }
75  }

16  package com.irurueta.numerical.roots;
17  
18  import com.irurueta.numerical.InvalidBracketRangeException;
19  import com.irurueta.numerical.LockedException;
20  import com.irurueta.numerical.NotAvailableException;
21  import com.irurueta.numerical.NotReadyException;
22  import com.irurueta.numerical.SingleDimensionFunctionEvaluatorListener;
23  import com.irurueta.statistics.UniformRandomizer;
24  import org.junit.jupiter.api.Test;
25  
26  import static org.junit.jupiter.api.Assertions.*;
27  
28  class FalsePositionSingleRootEstimatorTest {
29  
30      public static final double MIN_EVAL_POINT = 0.0;
31      public static final double MAX_EVAL_POINT = 1.0;
32  
33      public static final double MIN_TOLERANCE = 3e-8;
34      public static final double MAX_TOLERANCE = 1e-5;
35  
36      private double constant;
37      private double root1;
38      private double root2;
39      private double root3;
40  
41      private final SingleDimensionFunctionEvaluatorListener constantPolynomial = point -> constant;
42      private final SingleDimensionFunctionEvaluatorListener firstDegreePolynomial = point -> (point - root1);
43      private final SingleDimensionFunctionEvaluatorListener secondDegreePolynomial =
44              point -> (point - root1) * (point - root2);
45      private final SingleDimensionFunctionEvaluatorListener secondDegreePolynomialWithTwoComplexConjugateRoots =
46              point -> (point * point + Math.abs(root1));
47      private final SingleDimensionFunctionEvaluatorListener thirdDegreePolynomial =
48              point -> (point - root1) * (point - root2) * (point - root3);
49      private final SingleDimensionFunctionEvaluatorListener thirdDegreePolynomialWithTripleRoot =
50              point -> (point - root1) * (point - root1) * (point - root1);
51  
52      @Test
53      void testConstructor() throws NotAvailableException, InvalidBracketRangeException {
54  
55          final var randomizer = new UniformRandomizer();
56          final var minEvalPoint = randomizer.nextDouble(MIN_EVAL_POINT, MAX_EVAL_POINT);
57          final var maxEvalPoint = randomizer.nextDouble(minEvalPoint, MAX_EVAL_POINT);
58          final var tolerance = randomizer.nextDouble(MIN_TOLERANCE, MAX_TOLERANCE);
59  
60          // Testing 1st constructor
61          var estimator = new FalsePositionSingleRootEstimator();
62          assertNotNull(estimator);
63  
64          assertThrows(NotAvailableException.class, estimator::getListener);
65          assertEquals(FalsePositionSingleRootEstimator.DEFAULT_MAX_EVAL_POINT, estimator.getMaxEvaluationPoint(),
66                  0.0);
67          assertEquals(FalsePositionSingleRootEstimator.DEFAULT_MIN_EVAL_POINT, estimator.getMinEvaluationPoint(),
68                  0.0);
69          assertThrows(NotAvailableException.class, estimator::getRoot);
70          assertEquals(FalsePositionSingleRootEstimator.DEFAULT_TOLERANCE, estimator.getTolerance(), 0.0);
71          assertTrue(estimator.isBracketAvailable());
72          assertFalse(estimator.isListenerAvailable());
73          assertFalse(estimator.isLocked());
74          assertFalse(estimator.isReady());
75          assertFalse(estimator.isRootAvailable());
76  
77          // Testing 2nd constructor
78          estimator = new FalsePositionSingleRootEstimator(constantPolynomial, minEvalPoint, maxEvalPoint, tolerance);
79          assertNotNull(estimator);
80  
81          assertEquals(constantPolynomial, estimator.getListener());
82          assertEquals(maxEvalPoint, estimator.getMaxEvaluationPoint(), 0.0);
83          assertEquals(minEvalPoint, estimator.getMinEvaluationPoint(), 0.0);
84          assertThrows(NotAvailableException.class, estimator::getRoot);
85          assertEquals(tolerance, estimator.getTolerance(), 0.0);
86          assertTrue(estimator.isBracketAvailable());
87          assertTrue(estimator.isListenerAvailable());
88          assertFalse(estimator.isLocked());
89          assertTrue(estimator.isReady());
90          assertFalse(estimator.isRootAvailable());
91  
92          // Force InvalidBracketRangeException
93          assertThrows(InvalidBracketRangeException.class, () -> new FalsePositionSingleRootEstimator(constantPolynomial,
94                  maxEvalPoint, minEvalPoint, tolerance));
95  
96          // Force IllegalArgumentException
97          assertThrows(IllegalArgumentException.class, () -> new FalsePositionSingleRootEstimator(constantPolynomial,
98                  minEvalPoint, maxEvalPoint, -tolerance));
99      }

101     @Test

103 
104         final var estimator = new FalsePositionSingleRootEstimator();
105 
106         // check default values

108         assertFalse(estimator.isListenerAvailable());
109         assertFalse(estimator.isReady());
110 
111         // set listener
112         estimator.setListener(constantPolynomial);
113         // check correctness
114         assertEquals(constantPolynomial, estimator.getListener());
115         assertTrue(estimator.isListenerAvailable());
116         assertTrue(estimator.isReady());
117     }

122

125         final var maxEvalPoint = randomizer.nextDouble(minEvalPoint, MAX_EVAL_POINT);
126 
127         final var estimator = new FalsePositionSingleRootEstimator();
128 
129         // check default values
130         assertTrue(estimator.isBracketAvailable());
131         assertEquals(FalsePositionSingleRootEstimator.DEFAULT_MIN_EVAL_POINT, estimator.getMinEvaluationPoint(),
132                 0.0);
133         assertEquals(FalsePositionSingleRootEstimator.DEFAULT_MAX_EVAL_POINT, estimator.getMaxEvaluationPoint(),
134                 0.0);
135

137         estimator.setBracket(minEvalPoint, maxEvalPoint);
138         // check correctness
139         assertTrue(estimator.isBracketAvailable());
140         assertEquals(minEvalPoint, estimator.getMinEvaluationPoint(), 0.0);
141         assertEquals(maxEvalPoint, estimator.getMaxEvaluationPoint(), 0.0);
142 
143         // Force InvalidBracketRangeException
144         assertThrows(InvalidBracketRangeException.class, () -> estimator.setBracket(maxEvalPoint, minEvalPoint));
145     }

167     @Test
168     void testEstimate() throws LockedException, NotReadyException, InvalidBracketRangeException,
169             RootEstimationException, NotAvailableException {
170         final var randomizer = new UniformRandomizer();
171         constant = randomizer.nextDouble(MIN_EVAL_POINT, MAX_EVAL_POINT);
172         root1 = randomizer.nextDouble(MIN_EVAL_POINT, 0.2 * MAX_EVAL_POINT);
173         root2 = randomizer.nextDouble(0.4 * MAX_EVAL_POINT, 0.6 * MAX_EVAL_POINT);
174         root3 = randomizer.nextDouble(0.8 * MAX_EVAL_POINT, MAX_EVAL_POINT);
175 
176         // instantiate estimator with brackets for accuracy (otherwise estimation
177         // might fail
178         final var estimator = new FalsePositionSingleRootEstimator();
179 
180         // test constant polynomial (has no root)
181         estimator.setListener(constantPolynomial);
182         assertFalse(estimator.isLocked());
183         assertThrows(RootEstimationException.class, () -> estimator.computeBracket(MIN_EVAL_POINT, MAX_EVAL_POINT));
184         assertFalse(estimator.isLocked());
185         assertThrows(RootEstimationException.class, estimator::estimate);
186         assertFalse(estimator.isLocked());
187         assertFalse(estimator.isRootAvailable());
188         assertThrows(NotAvailableException.class, estimator::getRoot);
189 
190         // reset bracket
191         estimator.setBracket(MIN_EVAL_POINT, MAX_EVAL_POINT);
192 
193         // test 1st degree polynomial
194         estimator.setListener(firstDegreePolynomial);
195         assertFalse(estimator.isLocked());
196         estimator.computeBracket(MIN_EVAL_POINT, MAX_EVAL_POINT);
197         assertFalse(estimator.isLocked());
198         estimator.estimate();
199         assertFalse(estimator.isLocked());
200         assertTrue(estimator.isRootAvailable());
201         assertEquals(estimator.getRoot(), root1, 2.0 * estimator.getTolerance());
202 
203         // reset bracket
204         estimator.setBracket(MIN_EVAL_POINT, MAX_EVAL_POINT);
205 
206         // test 2nd degree polynomial
207         estimator.setListener(secondDegreePolynomial);
208         assertFalse(estimator.isLocked());
209         estimator.computeBracket(MIN_EVAL_POINT, 0.5 * (root1 + root2));
210         assertFalse(estimator.isLocked());
211         estimator.estimate();
212         assertFalse(estimator.isLocked());
213         assertTrue(estimator.isRootAvailable());
214         assertEquals(estimator.getRoot(), root1, 2.0 * estimator.getTolerance());
215 
216         assertFalse(estimator.isLocked());
217         estimator.computeBracket(0.5 * (root1 + root2), 0.5 * (root2 + root3));
218         assertFalse(estimator.isLocked());
219         estimator.estimate();
220         assertFalse(estimator.isLocked());
221         assertTrue(estimator.isRootAvailable());
222         assertEquals(estimator.getRoot(), root2, 2.0 * estimator.getTolerance());
223 
224         // reset bracket
225         estimator.setBracket(MIN_EVAL_POINT, MAX_EVAL_POINT);
226 
227         // test 2nd degree polynomial with two complex conjugate roots
228         estimator.setListener(secondDegreePolynomialWithTwoComplexConjugateRoots);
229         assertFalse(estimator.isLocked());
230         assertThrows(RootEstimationException.class, () -> estimator.computeBracket(MIN_EVAL_POINT, MAX_EVAL_POINT));
231         assertFalse(estimator.isLocked());
232         assertThrows(RootEstimationException.class, estimator::estimate);
233         assertFalse(estimator.isLocked());
234         assertFalse(estimator.isRootAvailable());
235 
236         // reset bracket
237         estimator.setBracket(MIN_EVAL_POINT, MAX_EVAL_POINT);
238 
239         // test 3rd degree polynomial
240         // we need to properly set bracketing for each root and then refine the
241         // result using estimate method
242         estimator.setListener(thirdDegreePolynomial);
243         assertFalse(estimator.isLocked());
244         estimator.computeBracket(MIN_EVAL_POINT, 0.5 * (root1 + root2));
245         assertFalse(estimator.isLocked());
246         estimator.estimate();
247         assertFalse(estimator.isLocked());
248         assertTrue(estimator.isRootAvailable());
249         assertEquals(estimator.getRoot(), root1, 2.0 * estimator.getTolerance());
250 
251         assertFalse(estimator.isLocked());
252         estimator.computeBracket(0.5 * (root1 + root2), 0.5 * (root2 + root3));
253         assertFalse(estimator.isLocked());
254         estimator.estimate();
255         assertFalse(estimator.isLocked());
256         assertTrue(estimator.isRootAvailable());
257         assertEquals(estimator.getRoot(), root2, estimator.getTolerance());
258 
259         assertFalse(estimator.isLocked());
260         estimator.computeBracket(0.5 * (root2 + root3), MAX_EVAL_POINT);
261         assertFalse(estimator.isLocked());
262         estimator.estimate();
263         assertFalse(estimator.isLocked());
264         assertTrue(estimator.isRootAvailable());
265         assertEquals(estimator.getRoot(), root3, 2.0 * estimator.getTolerance());
266 
267         // reset bracket
268         estimator.setBracket(MIN_EVAL_POINT, MAX_EVAL_POINT);
269 
270         // reset bracket
271         estimator.setBracket(MIN_EVAL_POINT, MAX_EVAL_POINT);
272 
273         // test 3rd degree polynomial with triple root
274         estimator.setListener(thirdDegreePolynomialWithTripleRoot);
275         assertFalse(estimator.isLocked());
276         estimator.computeBracket(0.9 * root1, 1.1 * root1);
277         assertFalse(estimator.isLocked());
278         estimator.estimate();
279         assertFalse(estimator.isLocked());
280         assertTrue(estimator.isRootAvailable());
281         assertEquals(estimator.getRoot(), root1, 2.0 * estimator.getTolerance());
282 
283         // reset bracket
284         estimator.setBracket(MIN_EVAL_POINT, MAX_EVAL_POINT);
285 
286         // test 3rd degree polynomial with 1 real root and 2 conjugate complex
287         // roots
288         estimator.setListener(thirdDegreePolynomialWithTripleRoot);
289         assertFalse(estimator.isLocked());
290         estimator.computeBracket(0.9 * root1, 1.1 * root1);
291         assertFalse(estimator.isLocked());
292         estimator.estimate();
293         assertFalse(estimator.isLocked());
294         assertTrue(estimator.isRootAvailable());
295         assertEquals(estimator.getRoot(), root1, 2.0 * estimator.getTolerance());
296     }
